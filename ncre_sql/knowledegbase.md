```sql
DEFAULT CHARACTER SET GB2312 ;-- 设置默认值

SHOW DATABASE; -- 查看数据库
ALTER DATABASE db_name DEFAULT CHARACTER SET gb2132--修改数据库
DROP DATABASE db_name--删除数据库
```
